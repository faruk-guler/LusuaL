ansibleshell () {
    /usr/bin/ansible all -m shell -a "$*"
}
ansibleshell () {
    if [[ "$1" =~ ^- ]] || [[ "$1" == "all" ]] || [[ "$1" == "localhost" ]]; then
        command ansible "$@"
    else
        command ansible all -m shell -a "$*"
    fi
}
